docker compose exec -T api python - <<'PY'
import os, sqlite3

db_path = os.environ.get("DB_PATH", "/data/app.db")
conn = sqlite3.connect(db_path)
conn.row_factory = sqlite3.Row

cols = [r["name"] for r in conn.execute("PRAGMA table_info(recon_cluster)").fetchall()]
print("DB_PATH =", db_path)
print("recon_cluster columns:", cols)

rows = conn.execute("SELECT * FROM recon_cluster LIMIT 2").fetchall()
print(f"\nFirst {len(rows)} rows:")
for i, r in enumerate(rows, 1):
    d = dict(r)
    print(f"\n--- row {i} ---")
    for k in cols:
        print(f"{k}: {d.get(k)}")

conn.close()
PY
